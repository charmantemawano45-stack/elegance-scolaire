      signupBtn.innerHTML = '<i class="fas fa-user-check"></i> S\'inscrire';

    initiatePaymentBtnModal.addEventListener('click', async () => {
      const amount = parseFloat(paymentAmount.value);
      if (isNaN(amount) || amount <= 0) { alert('Montant invalide'); return; }
      const remaining = DEFAULT_FEES - paidAmount;
      if (amount > remaining) { alert('Montant supérieur au solde dû'); return; }
      
      const method = paymentMethod.value;
      const phone = phoneNumber.value.trim();
      if (!phone) { alert('Numéro de téléphone requis'); return; }
      
      // Validation du numéro selon l'opérateur
      const phoneRegex = /^\+243(\d{9})$/;
      const phoneMatch = phone.match(phoneRegex);
      if (!phoneMatch) {
        alert('Format de numéro invalide. Utilisez +243 suivi de 9 chiffres.');
        return;
      }
      const prefix = phoneMatch[1].substring(0, 2);
      let validPrefixes = [];
      if (method === 'Illicocash' || method === 'Orange Money') {
        validPrefixes = ['89', '85', '84'];
      } else if (method === 'M-Pesa') {
        validPrefixes = ['81', '82', '83', '86'];
      }
      if (!validPrefixes.includes(prefix)) {
        alert(`Numéro non valide pour ${method}. Préfixes acceptés : ${validPrefixes.join(', ')}`);
        return;
      }
      
      paymentStep1.style.display = 'none';
      paymentStep2.style.display = 'block';
      paymentProgress.innerHTML = 'Connexion à la passerelle mobile money...';
      
      for (let i = 1; i <= 10; i++) {
        await new Promise(r => setTimeout(r, 1000));
        paymentProgress.innerHTML = `Étape ${i}/10 : ${i<5?'Authentification':'Préparation transaction'}...`;
      }
      
      pendingPaymentData = { amount, method, phone };
      paymentModal.style.display = 'none';
      
      confirmAmountSpan.textContent = amount.toFixed(2);
      confirmMethodSpan.textContent = method;
      mobileConfirmModal.style.display = 'flex';
      mobilePassword.value = '';
    });

          <p>Transaction: ${receiptData.transaction}</p>
